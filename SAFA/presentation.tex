Let $|G| < \infty$ and $V=\mathbb{C}^n$.
\begin{definition}
    $\rho  \colon G  \to \text{GL}(V)$ is \emph{irreducalbe}, if there is no $0 < W < V$, such that $G \cdot W \leq W$.
\end{definition}
\begin{theorem}
    Every representation $\rho \colon G \to GL(V)$ is equal to 
    $ \oplus  _{i \in I} \rho_i$, where every $\rho_i$ is irreducable.
\end{theorem}

\begin{frame}{\color{aoblue}Modeling representations}
\note{
Explain Urban's idea - start with an arbitrary function $G \to \mathbb{C}^{n \times n}$ and change its parameter (with gradient descent) (Explanation on how to change the parameters on the following slides!)
}
    Take any model 
    \begin{align*}
    \hat{\rho} \colon G &\to \mathbb{C}^{n \times n} \\    
    g &\mapsto \begin{bmatrix}
        g_{1,1} & \cdots & g_{1,n} \\
        \vdots & & \vdots \\ 
        g_{n,1} & \cdots & g_{n,n}
    \end{bmatrix} = \phi_g 
    \end{align*}
and keep changing its parameters $\phi = \{\phi_g \mid g \in G\}$ until $\hat{\rho}$ becomes a representation.
\end{frame}
\begin{frame}{Just on generators}
\note{
In practice, we want small number of parameters --> define just on generators.
}

Let $G = <S | R>$. Define $\hat{\rho} \colon G \to \mathbb{C}^{n \times n}  $ as 
    $$
    \hat{\rho}(s) =\begin{bmatrix}
        s_{1,1} & \cdots & s_{1,n} \\
        \vdots & & \vdots \\

    \end{bmatrix} = \phi_s
    $$ on generators $s \in S$ and

\begin{frame}{\color{aoblue}Relation loss}
\note{
\textbf{When will $\hat{\rho}$  be homomomorphism? Use the relations, define loss funcion, explain how it "measures, how close one is to a homomorphism"}
}
Let $G =<S|R>$.
    $\hat \rho $ is a homomorphism iff $$\hat\rho(r) - I = 0$$ for every $r \in R$.

    \begin{equation*}
        \loss_{\text{rel}}(\hat\rho) = \frac{1}{|R|} \sum \limits_{r \in R}  ||\hat \rho(r) - I  ||_F^2
    \end{equation*}
    \begin{theorem}
        $\hat\rho$ is homomorphism iff $\loss_{\text{rel}}(\hat\rho) = 0$
    \end{theorem}
\end{frame}
\begin{frame}{\color{aoblue}Gradient descent}
\note{
We will try to minimise loss with gradient descent. 
Examplain  on whiteboard that our model $\hat \rho$ is model in parameters $\phi = \{\phi_s \mid s \in S\}$. 
Explain that we will try to minimise this loss by taking small steps in the oposite direction of the gradient.
}
Optimisation using
    $$
    \phi_{i+1} = \phi_i - \eta \nabla \loss(\phi_i) 
    $$
    Gradient flow equation
    $$
    \frac{d\phi}{dt} = - \nabla \loss(\phi), \quad \phi(0) = \phi_0
    $$
\end{frame}

\begin{frame}{\color{aoblue}Irreducibility loss}
    \begin{theorem}
    $\rho$ is irreducible iff $|\chi_\rho| = \frac{1}{|G|} 
    \sum\limits_{g \in G} \text{tr}(\rho(g)) \text{tr}(\rho(g^{-1}) = 1$.
\end{theorem}
\vfill
$$
\loss_{\text{irr}}(\hat \rho) = (|\chi_{\hat \rho}| - 1 )^2
$$
\end{frame}
\begin{frame}{\color{aoblue}Unitary loss}
    \begin{theorem}
       Every representation $\rho \colon G \to GL_n$ of finite group is equivalent to unitary representation $\rho_{\text{unitary}} \colon G \to U_n$.  
    \end{theorem}
    $$
    \loss_{\text{unitary}} = \frac{1}{|S|} \sum \limits_{s \in S} ||\hat\rho(s) \hat\rho(s)^H - I||
    $$
    \vfill
    $$
    \loss = \loss_\text{rel}+\loss_\text{irr}+\loss_\text{unitary}
    $$
\end{frame}

D_{2n} = \langle r, s \mid r^n = s^2 = (rs)^2 = 1 \rangle

\begin{theorem}
The irreducible representations of the dihedral group \( D_{2n} \) are:
\begin{columns}

            \rho_k \colon  s &\mapsto \begin{bmatrix}

            r &\mapsto \delta

\end{columns}

where \( 0 \leq k < \frac{n}{2} \), \( \varepsilon \in \{-1, 1\} \), and

\begin{frame}{\color{aoblue} Group actions}
    \note{
Lets adapt the approach to group actions.

Show some motivation on the board. 

Discrete space is not suitable for GD - work with probabilities!
    Explain how we get to the matrices, explain that this is not the only way
    }
  Map generators of $G=<S|R>$ to random maps on $[n]$.
    \begin{align*}
        \hat \rho \colon S &\to \mathcal{P}(\operatorname{fun}([n],[n])) \\
    s &\mapsto P_s = \begin{bmatrix}
        P(s(1) = 1) & P(s(1) = 2) & \dotsm & P(s(1) = n) \\
        P(s(2) = 1) & P(s(2) = 2) & \dotsm & P(s(2) = n) \\
        \vdots & \vdots & & \vdots \\
        P(s(n) = 1) & P(s(n) = 2) & \dotsm & P(s(n) = n) \\
    \end{bmatrix}
    \end{align*}
    For $f \in \operatorname{fun}([n],[n])$, define $P(s = f) = \prod \limits_{i=1}^n P(s(i) = f(i)) = \prod \limits_{i=1}^n s_{i, f(i)}$
\end{frame}
\begin{frame}{Construction details}
    For $g=s_1s_2\dotsm s_m \in G$ define 
    $$P_g = P_{s_1} P_{s_2} \dotsm P_{s_m} = \begin{bmatrix}
          P(g(1) = 1) & \dotsm & P(g(1) = n) \\
        \vdots & \vdots & & \vdots \\
        P(g(n) = 1) & \dotsm & P(g(n) = n) \\
    \end{bmatrix}$$ 
    Get stochastic matrices using $\operatorname{softmax}(v)_i = \frac{e^{v_i}}{\sum_{j=1}^n e^{v_j}}$
    \begin{align*}
        P \colon \mathbb{R}^{n \times n} & \to \mathcal{S}_n \\
        \begin{bmatrix}
            a_1^T \\
            \vdots \\
            a_n^T
        \end{bmatrix} &\mapsto 
        \begin{bmatrix}
            \operatorname{softmax}(a_1^T) \\
            \vdots \\
            \operatorname{softmax}(a_n^T)
        \end{bmatrix} 
    \end{align*}
\end{frame}
\begin{frame}{\color{aoblue} Make relations probable}
 For $r \in R$, we want $P(r=\operatorname{id}) = 1$
 $$
 0 = \log(P(r=\operatorname{id})) = \log(\prod_{i=1}^n P(r(i)=i)) = \operatorname{tr}(\log(P_r))
 $$
 $$
 \mathcal{L}_{\text{rel}}  = - \frac{1}{|R|} \sum_{r \in R} \operatorname{tr}(\log(P_r))
 $$
\end{frame}
\begin{frame}{\color{aoblue} Make bijections probable}
    \note{Similarly to the unitarity constraint, we can constraint bijectivity}
    $$
    P(s \in S_n )= \sum_{\sigma \in S_n} P(s = \sigma) = \sum_{\sigma \in S_n}   \prod_{i=1}^n s_{i, \sigma(i)}
= \text{Perm} (P_s).
    $$
    $P(s \in S_n ) = 1$ $\iff$ $P_S$ is permutation matrix $\iff$ $P_S \in U_n$
    \vfill
    $$
    \mathcal{L}_{\text{bijective}} = \frac{1}{|S|} \sum_{s \in S}||P_sP_s^T - I||_F^2  = \mathcal{L}_{\text{unitary}} \circ \operatorname{softmax}
    $$
\end{frame}

\begin{frame}{\color{aoblue} Graph isomorphisms}
\begin{itemize}
    \item $G_1, G_2$ graphs with adjacency matrices $M_1 = \begin{bmatrix}
        m_{i,j}^{(1)}
    \end{bmatrix}$ and $M_2 = \begin{bmatrix}
        m_{i,j}^{(1)}
    \end{bmatrix}$

    \item Random mapping $f \colon [n] \to [n] \sim P_f = \begin{bmatrix}P(f(i)=j)
    \end{bmatrix}$

    
    \item For every $i \sim_1 j$, we want $P(f(i) \sim_2 f(j) = 1$ 
\end{itemize}    

\end{frame}
\begin{frame}{\color{aoblue}Relation loss}
 
$$
P(f(i) \sim_2 f(j)) = \sum_{k= 1}^n \sum_{h = 1}^n f_{i, k} f_{j, h} m_{k,h}^{(2)} 
% = f_j^T M_1 f_i
= (P M_2 P^T)_{i, j}
$$  
\vfill
\begin{align*}
    \mathcal L_{\text{rel}}  &= -\sum_{i \sim_1 j} P(f(i) \sim_2 f(j)) 
    \\&= -\sum_{i=1}^n\sum_{j=1}^n log(f_j^T M_2 f_i)m_{i,j}^{(1)}\\&= -\operatorname{tr}(\log(P_fM_2P_f^T) M_1^T)
\end{align*}
\end{frame}
