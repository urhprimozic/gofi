\begin{frame}{\color{aoblue}Diskretni objekti so ničle funkcionalov}
\note{
Razloži osnovno idejo. Matematične objekte vpeno v evklidkski prostor in jih enačimo z množico ničel 
nekega funkcionala. Potem lahko te objekte iščemo preko optimizacije.
}
    \begin{itemize}
        \item iščemo objekte $Z  \subset Y^X$
        \item vpetje  $Y^X \hookrightarrow  \R^N$
        \item $Z = \loss^{-1}(0)$ za $\loss \colon \R^N \to [0,\infty)$
        \item $\operatorname{argmin}\loss \in Z$
    \end{itemize}
\end{frame}
