\begin{frame}{\color{aoblue}Ciklične grupe}
    \note{
        $C_n$ ima le en generator $z$, vse nerazcepne upodobitve so enodimenzionalne in kompleksne. 
        Torej iščeš $\hat\rho(z)=x+iy \in  \C$.
    }
    Nerazcepne
upodobitve $C_n=\langle z \mid z^n = 1 \rangle$ so
$$
\rho_k(z) = e^{\frac{2 \pi i k}{n}}.
$$
Funkcije izgube:
\begin{align*}
      \Loss{rel} &= |(x + iy)^n -1|^2\\
  \Loss{irr} &= (\frac{1}{n} \sum_{j=0}^{n-1} |(x + iy)^j|^2 -1)^2 \\
  \Loss{unitary} &= | x^2 + y^2 -1|^2.
\end{align*}
\end{frame}

\begin{frame}{\color{aoblue}Diedrske grupe}
    $D_n = \langle r, s \mid r^n, s^2, (rs)^2 \rangle$
\begin{align*}
  \chi_{\varepsilon, \delta} \colon \quad
  s &\mapsto \varepsilon, \\
  r &\mapsto \delta, \quad  \varepsilon \in \{-1, 1\}, \delta \in \{1, -1^{n + 1}\}
\end{align*}
\begin{align*}
  \rho_k \colon  s &\mapsto
  \begin{bmatrix}
    0 & 1 \\
    1 & 0
  \end{bmatrix}\\
  r &\mapsto
  \begin{bmatrix}
    \cos\left(\frac{2\pi k}{n}\right) & -\sin\left(\frac{2\pi
    k}{n}\right) \\
    \sin\left(\frac{2\pi k}{n}\right) & \cos\left(\frac{2\pi k}{n}\right)
  \end{bmatrix}
\end{align*}
\end{frame}
